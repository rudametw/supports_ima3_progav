//Exemple
int recherche(int x, Liste l) {
	int existe ; Ptcellule p;
	p = l;
	while ( (p != NULL) && (p->valeur != x) ) {
		p = p->suivant;
	}
	existe = (p!=NULL);
	return (existe);
}



void ajout_tete (int x, Liste *pL){
	/* pL en D/R */
	Ptcellule p;
	p = allouer();
	p->valeur = x;
	p->suivant = *pL;
	*pL = p;
}
